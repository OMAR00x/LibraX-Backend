// ========================================
// 🧪 اختبار Login - نسخ الكود في Tinker
// ========================================
// php artisan tinker
// ثم انسخ الأوامر التالية واحد تلو الآخر
// ========================================

// 1️⃣ إنشاء مستخدم تجريبي
$user = \App\Models\User::create(['first_name' => 'أحمد', 'last_name' => 'محمد', 'phone' => '966500000001', 'password' => \Hash::make('password123'), 'role' => 'customer', 'is_active' => true]);

// 2️⃣ اختبار Login ناجح
$response = \Http::post('http://127.0.0.1:8000/api/login', ['phone' => '966500000001', 'password' => 'password123']);
echo "Status: {$response->status()}\n"; print_r($response->json());

// 3️⃣ اختبار كلمة مرور خاطئة
$response = \Http::post('http://127.0.0.1:8000/api/login', ['phone' => '966500000001', 'password' => 'wrong']);
echo "Status: {$response->status()}\n"; print_r($response->json());

// 4️⃣ اختبار حساب غير موجود
$response = \Http::post('http://127.0.0.1:8000/api/login', ['phone' => '966500000099', 'password' => 'password123']);
echo "Status: {$response->status()}\n"; print_r($response->json());

// 5️⃣ اختبار حساب غير فعال
$user->update(['is_active' => false]);
$response = \Http::post('http://127.0.0.1:8000/api/login', ['phone' => '966500000001', 'password' => 'password123']);
echo "Status: {$response->status()}\n"; print_r($response->json());

// 6️⃣ تنظيف - حذف المستخدم
$user->forceDelete();
echo "✅ تم حذف المستخدم التجريبي\n";
